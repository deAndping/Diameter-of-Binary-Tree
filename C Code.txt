/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
int max(int a, int b){
    return a > b?   a : b;
}

int pathLength(struct TreeNode* root){
    if(!root)   return 0;
    else    return max(1 + pathLength(root->left), 1 + pathLength(root->right));
}

int diameterOfBinaryTree(struct TreeNode* root) {
    if(!root)   return 0;
    int len = pathLength(root->left) + pathLength(root->right);
    return  max(len, max(diameterOfBinaryTree(root->left), diameterOfBinaryTree(root->right)));
}
